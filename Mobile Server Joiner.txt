-- Function to load servers
local function LoadServers()
    -- Clear existing
    for _, child in pairs(ScrollFrame:GetChildren()) do
        if child:IsA("Frame") then
            child:Destroy()
        end
    end

    StatusLabel.Text = "Loading servers..."
    local servers = GetServerList:InvokeServer("Main")
    local count = 0

    for key, server in pairs(servers) do
        if typeof(server) == "table" and server.PlatformType == "MobileOnly" then
            count = count + 1

            local ServerBtn = Instance.new("Frame")
            ServerBtn.Size = UDim2.new(1, 0, 0, 60)
            ServerBtn.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
            ServerBtn.BorderSizePixel = 0
            ServerBtn.Parent = ScrollFrame

            local BtnCorner = Instance.new("UICorner")
            BtnCorner.CornerRadius = UDim.new(0, 8)
            BtnCorner.Parent = ServerBtn

            local NameLabel = Instance.new("TextLabel")
            NameLabel.Size = UDim2.new(0.65, 0, 0, 25)
            NameLabel.Position = UDim2.new(0, 10, 0, 5)
            NameLabel.BackgroundTransparency = 1
            NameLabel.Text = server.Name
            NameLabel.TextColor3 = Color3.fromRGB(200, 80, 20)
            NameLabel.TextSize = 16
            NameLabel.Font = Enum.Font.GothamBold
            NameLabel.TextXAlignment = Enum.TextXAlignment.Left
            NameLabel.Parent = ServerBtn

            local InfoLabel = Instance.new("TextLabel")
            InfoLabel.Size = UDim2.new(0.65, 0, 0, 20)
            InfoLabel.Position = UDim2.new(0, 10, 0, 30)
            InfoLabel.BackgroundTransparency = 1
            InfoLabel.Text = tostring(server.Region) .. " | " .. tostring(server.Size)
            InfoLabel.TextColor3 = Color3.fromRGB(150, 150, 150)
            InfoLabel.TextSize = 13
            InfoLabel.Font = Enum.Font.Gotham
            InfoLabel.TextXAlignment = Enum.TextXAlignment.Left
            InfoLabel.Parent = ServerBtn

            local JoinBtn = Instance.new("TextButton")
            JoinBtn.Size = UDim2.new(0, 70, 0, 35)
            JoinBtn.Position = UDim2.new(1, -80, 0.5, -17)
            JoinBtn.BackgroundColor3 = Color3.fromRGB(200, 80, 20)
            JoinBtn.Text = "JOIN"
            JoinBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
            JoinBtn.TextSize = 14
            JoinBtn.Font = Enum.Font.GothamBold
            JoinBtn.BorderSizePixel = 0
            JoinBtn.Parent = ServerBtn

            local JoinCorner = Instance.new("UICorner")
            JoinCorner.CornerRadius = UDim.new(0, 6)
            JoinCorner.Parent = JoinBtn

            local serverKey = key
            JoinBtn.MouseButton1Click:Connect(function()
                JoinBtn.Text = "..."
                PickGameMode:FireServer("Main", nil, serverKey)
            end)
        end
    end

    ScrollFrame.CanvasSize = UDim2.new(0, 0, 0, count * 65)
    StatusLabel.Text = "Found " .. count .. " mobile servers"
end

RefreshBtn.MouseButton1Click:Connect(LoadServers)

-- Make draggable
local dragging, dragStart, startPos
MainFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStart = input.Position
        startPos = MainFrame.Position
    end
end)

MainFrame.InputChanged:Connect(function(input)
    if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
        local delta = input.Position - dragStart
        MainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)

MainFrame.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = false
    end
end)

-- Load on start
LoadServers()
